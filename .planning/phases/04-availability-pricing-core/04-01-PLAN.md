    ```typescript
    import type { db } from "@my-app/db";
    import type { listingAvailabilityRule, listingAvailabilityBlock, listingAvailabilityException } from "@my-app/db/schema/availability";
    export type Db = typeof db;
    export type AvailabilityRuleRow = typeof listingAvailabilityRule.$inferSelect;
    export type AvailabilityBlockRow = typeof listingAvailabilityBlock.$inferSelect;
    export type AvailabilityExceptionRow = typeof listingAvailabilityException.$inferSelect;

    export interface CreateAvailabilityRuleInput {
      listingId: string;
      organizationId: string; // used to verify listing ownership via alias check
      dayOfWeek: number;       // 0–6
      startMinute: number;     // 0–1440
      endMinute: number;       // must be > startMinute
    }
    export interface CreateAvailabilityBlockInput {
      listingId: string;
      organizationId: string;
      startsAt: Date;
      endsAt: Date;
      reason?: string;
    }
    export interface CreateAvailabilityExceptionInput {
      listingId: string;
      organizationId: string;
      date: Date;
      isAvailable: boolean;
      startMinute?: number;
      endMinute?: number;
      reason?: string;
    }
    ```

    **src/availability-service.ts** — all functions accept `db: Db` as last parameter:

    - `createAvailabilityRule(input: CreateAvailabilityRuleInput, db: Db): Promise<AvailabilityRuleRow>`
      — Verify listing belongs to org: select listing where `id = input.listingId AND organizationId = input.organizationId`; if not found throw `Error("NOT_FOUND")`.
      — Insert `listingAvailabilityRule` with `crypto.randomUUID()` id.

    - `deleteAvailabilityRule(id: string, organizationId: string, db: Db): Promise<void>`
      — Delete rule where `id = id AND listingId IN (SELECT id FROM listing WHERE organizationId = organizationId)`. If no rows deleted, throw `Error("NOT_FOUND")`.
      — Implementation: first fetch the rule, then verify the listing's org, then delete.

    - `listAvailabilityRules(listingId: string, organizationId: string, db: Db): Promise<AvailabilityRuleRow[]>`
      — Verify listing ownership, then select all rules for `listingId` ordered by `dayOfWeek, startMinute`.

    - `createAvailabilityBlock(input: CreateAvailabilityBlockInput, db: Db): Promise<AvailabilityBlockRow>`
      — Verify listing ownership; insert block with `source: "manual"`.

    - `deleteAvailabilityBlock(id: string, organizationId: string, db: Db): Promise<void>`
      — Verify via listing lookup, then delete; throw `Error("NOT_FOUND")` if no match.

    - `createAvailabilityException(input: CreateAvailabilityExceptionInput, db: Db): Promise<AvailabilityExceptionRow>`
      — Verify listing ownership; try insert; on unique constraint violation (same listingId+date) throw `Error("DUPLICATE_DATE")`.

    - `deleteAvailabilityException(id: string, organizationId: string, db: Db): Promise<void>`
      — Verify via listing lookup, then delete; throw `Error("NOT_FOUND")` if no match.

    - `checkSlotAvailable(listingId: string, startsAt: Date, endsAt: Date, db: Db): Promise<boolean>`
      — Query booking table for any booking where:
        `listingId = listingId AND status NOT IN ('cancelled') AND startsAt < endsAt AND endsAt > startsAt`
        (i.e., `booking.startsAt < input.endsAt AND booking.endsAt > input.startsAt`)
      — Also query `listingAvailabilityBlock` for any active block where `isActive = true` and intervals overlap.
      — Returns `false` if any overlap found, `true` otherwise.

    - `assertSlotAvailable(listingId: string, startsAt: Date, endsAt: Date, db: Db): Promise<void>`
      — Calls `checkSlotAvailable`; throws `Error("SLOT_UNAVAILABLE")` if result is false.

    **src/index.ts** — barrel export of everything from types and availability-service.
