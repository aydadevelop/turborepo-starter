<!-- How to query published listings — join pattern for Drizzle: -->
```typescript
// The join that makes a listing "published on marketplace":
// listing.isActive = true
// listingPublication.isActive = true AND channelType = "platform_marketplace"
// Use Drizzle innerJoin or where with subquery:
import { and, eq, ilike, count } from "drizzle-orm";
import { listing, listingPublication, listingAsset } from "@my-app/db/schema/marketplace";

const rows = await db
  .select({ /* listing fields + primary asset key */ })
  .from(listing)
  .innerJoin(listingPublication, and(
    eq(listingPublication.listingId, listing.id),
    eq(listingPublication.isActive, true),
    eq(listingPublication.channelType, "platform_marketplace"),
  ))
  .leftJoin(listingAsset, and(
    eq(listingAsset.listingId, listing.id),
    eq(listingAsset.isPrimary, true),
    eq(listingAsset.kind, "image"),
  ))
  .where(and(
    eq(listing.isActive, true),
    input.type ? eq(listing.listingTypeSlug, input.type) : undefined,
    input.q ? ilike(listing.name, `%${input.q}%`) : undefined,
  ))
  .orderBy(listing.createdAt)
  .limit(input.limit ?? 20)
  .offset(input.offset ?? 0);
```

  import { createQuery } from "@tanstack/svelte-query";
  import { page } from "$app/state";
  import { orpc } from "$lib/orpc";

  // Reactive URL params → query options
  const listingsQuery = createQuery(() => ({
    ...orpc.storefront.list.queryOptions({
      type: $page.url.searchParams.get("type") ?? undefined,
      q: $page.url.searchParams.get("q") ?? undefined,
    }),
  }));
