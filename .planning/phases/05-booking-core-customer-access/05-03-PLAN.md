```typescript
import { ORPCError } from "@orpc/server";
import { db } from "@my-app/db";
import { domainFn, type DomainRow } from "@my-app/domain";
import { organizationPermissionProcedure, publicProcedure } from "../index";

export const xRouter = {
  action: organizationPermissionProcedure({ resource: ["create"] })
    .contract.procedure.handler(async ({ context, input }) => {
      // thin: call service, translate errors, format dates
      try {
        const row = await domainFn({ ...input, organizationId: context.activeMembership.organizationId }, db);
        return { ...row, createdAt: row.createdAt.toISOString(), updatedAt: row.updatedAt.toISOString() };
      } catch (e) {
        if (e instanceof Error && e.message === "NOT_FOUND") throw new ORPCError("NOT_FOUND");
        throw e;
      }
    }),
};
```
