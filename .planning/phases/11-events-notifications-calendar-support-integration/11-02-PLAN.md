<objective>
Expose customer-facing support history alongside own bookings on the live web surface.

Purpose: AUTH-02 requires a customer to view only their own bookings and support history across organizations they have booked with. The booking list API (`listMyBookings`) already exists. What is missing is: (1) a customer-scoped support list endpoint, and (2) a web page that displays both together. OPER-02 requires customers to be able to follow a support conversation tied to a booking.

Output:
- `listCustomerTickets(customerId, db)` domain function in `packages/support` — no org scoping, scoped by `customerUserId`
- `listMyTickets` contract route and `protectedProcedure` handler that calls it using session user ID
- `/dashboard/bookings` web page listing the customer's own bookings with linked support tickets (subject + status)
</objective>

```typescript
export async function listCustomerTickets(
  customerUserId: string,
  filter: ListTicketsFilter,
  db: Db,
): Promise<SupportTicketRow[]> {
  const conditions = [eq(supportTicket.customerUserId, customerUserId)];

  if (filter.status) {
    conditions.push(eq(supportTicket.status, filter.status));
  }
  if (filter.bookingId) {
    conditions.push(eq(supportTicket.bookingId, filter.bookingId));
  }

  const query = db
    .select()
    .from(supportTicket)
    .where(and(...conditions))
    .orderBy(supportTicket.createdAt);

  if (filter.limit !== undefined) {
    query.limit(filter.limit);
  }
  if (filter.offset !== undefined) {
    query.offset(filter.offset);
  }

  return query;
}
```

```svelte
<script lang="ts">
  import { createQuery } from "@tanstack/svelte-query";
  import { orpc } from "$lib/orpc";

  const bookingsQuery = createQuery(() =>
    orpc.booking.listMyBookings.queryOptions({ input: {} })
  );

  const ticketsQuery = createQuery(() =>
    orpc.support.listMyTickets.queryOptions({ input: {} })
  );

  // index tickets by bookingId for O(1) lookup
  const ticketsByBookingId = $derived(
    (ticketsQuery.data ?? []).reduce(
      (acc, t) => {
        if (t.bookingId) acc[t.bookingId] = t;
        return acc;
      },
      {} as Record<string, (typeof ticketsQuery.data)[number]>
    )
  );
</script>

<div class="container max-w-3xl py-8">
  <h1 class="text-2xl font-semibold mb-6">My Bookings</h1>

  {#if bookingsQuery.isPending}
    <p class="text-muted-foreground">Loading bookings…</p>
  {:else if bookingsQuery.isError}
    <p class="text-destructive">Failed to load bookings.</p>
  {:else if bookingsQuery.data?.length === 0}
    <p class="text-muted-foreground">You have no bookings yet.</p>
  {:else}
    <ul class="flex flex-col gap-4">
      {#each bookingsQuery.data ?? [] as booking (booking.id)}
        {@const ticket = ticketsByBookingId[booking.id]}
        <li class="rounded-lg border p-4">
          <div class="flex items-start justify-between gap-2">
            <div>
              <p class="font-medium text-sm">Booking #{booking.id.slice(0, 8)}</p>
              <p class="text-sm text-muted-foreground">
                {new Date(booking.startsAt).toLocaleDateString()} –
                {new Date(booking.endsAt).toLocaleDateString()}
              </p>
            </div>
            <span class="text-xs px-2 py-0.5 rounded border font-medium capitalize">
              {booking.status}
            </span>
          </div>

          {#if ticket}
            <div class="mt-3 border-t pt-3">
              <p class="text-sm font-medium">Support: {ticket.subject}</p>
              <p class="text-xs text-muted-foreground capitalize">{ticket.status.replace(/_/g, " ")}</p>
            </div>
          {/if}
        </li>
      {/each}
    </ul>
  {/if}
</div>
```
