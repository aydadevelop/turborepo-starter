Using `createStep` + `createWorkflow` from `@my-app/workflows`, scaffold `processCancellationWorkflow`:

```typescript
import { createStep, createWorkflow } from "@my-app/workflows";
import { updateBookingStatus } from "@my-app/booking";
import { emitDomainEvent } from "@my-app/events";
// import { initiateRefund } from "@my-app/payment";  — use when payment package exports this
import { evaluateCancellationPolicy } from "./cancellation-policy-service";

const evaluatePolicyStep = createStep("evaluate-cancellation-policy", async (ctx, input) => {
  const decision = evaluateCancellationPolicy(input);
  return { ...input, decision };
});

const applyRefundStep = createStep("apply-refund", async (ctx, input) => {
  // Call packages/payment initiateRefund when that function is available
  // For now: record the refund intent in bookingRefund table using db from ctx
  const db = ctx.db as Db;
  // Insert bookingRefund row with status: pending
  const [refund] = await db.insert(bookingRefund).values({
    id: crypto.randomUUID(),
    bookingId: input.bookingId,
    amountKopeks: input.decision.suggestedRefundKopeks,
    status: "pending",
  }).returning();
  return { ...input, refundId: refund.id };
}, {
  compensate: async (ctx, input) => {
    // Reverse the refund record if downstream step fails
    const db = ctx.db as Db;
    await db.update(bookingRefund)
      .set({ status: "reversed" })
      .where(eq(bookingRefund.id, input.refundId));
  }
});

const updateBookingCancelledStep = createStep("update-booking-cancelled", async (ctx, input) => {
  const db = ctx.db as Db;
  await updateBookingStatus({ id: input.bookingId, organizationId: input.organizationId, status: "cancelled" }, db);
  return input;
});

const emitBookingCancelledStep = createStep("emit-booking-cancelled", async (ctx, input) => {
  await emitDomainEvent({
    type: "booking:cancelled",
    organizationId: input.organizationId,
    idempotencyKey: `cancellation:${input.bookingId}`,
    data: { bookingId: input.bookingId, reason: input.reason ?? "cancellation_policy", refundAmountKopeks: input.decision.suggestedRefundKopeks },
  });
  return input;
});

export const processCancellationWorkflow = createWorkflow(
  "process-cancellation",
  [evaluatePolicyStep, applyRefundStep, updateBookingCancelledStep, emitBookingCancelledStep]
);
```

**Create `packages/disputes/src/dispute-workflow.ts`**:

```typescript
import { createStep, createWorkflow } from "@my-app/workflows";
import { emitDomainEvent } from "@my-app/events";
// DB: insert bookingDispute row

const openDisputeStep = createStep("open-dispute", async (ctx, input) => {
  const db = ctx.db as Db;
  const [dispute] = await db.insert(bookingDispute).values({
    id: crypto.randomUUID(),
    bookingId: input.bookingId,
    organizationId: input.organizationId,
    status: "open",
    openedAt: new Date(),
  }).returning();
  await emitDomainEvent({
    type: "dispute:opened",
    organizationId: input.organizationId,
    idempotencyKey: `dispute:opened:${dispute.id}`,
    data: { disputeId: dispute.id, bookingId: input.bookingId },
  });
  return { ...input, disputeId: dispute.id };
});

const resolveDisputeStep = createStep("resolve-dispute", async (ctx, input) => {
  const db = ctx.db as Db;
  await db.update(bookingDispute)
    .set({ status: "resolved", resolvedAt: new Date(), resolution: input.resolution })
    .where(eq(bookingDispute.id, input.disputeId));
  await emitDomainEvent({
    type: "dispute:resolved",
    organizationId: input.organizationId,
    idempotencyKey: `dispute:resolved:${input.disputeId}`,
    data: { disputeId: input.disputeId, resolution: input.resolution },
  });
  return input;
});

export const processDisputeWorkflow = createWorkflow(
  "process-dispute",
  [openDisputeStep, resolveDisputeStep]
);
```
