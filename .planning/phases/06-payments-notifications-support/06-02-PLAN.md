```typescript
import { eq, and, desc } from "drizzle-orm";
import { supportTicket, supportTicketMessage } from "@my-app/db/schema/support";
import type {
  Db,
  SupportTicketRow,
  SupportTicketMessageRow,
  CreateTicketInput,
  AddMessageInput,
  ListTicketsFilter,
} from "./types";

export async function createSupportTicket(
  input: CreateTicketInput,
  db: Db,
): Promise<SupportTicketRow> {
  const [row] = await db
    .insert(supportTicket)
    .values({
      id: crypto.randomUUID(),
      organizationId: input.organizationId,
      bookingId: input.bookingId,
      customerUserId: input.customerUserId,
      createdByUserId: input.createdByUserId,
      subject: input.subject,
      description: input.description,
      priority: input.priority ?? "normal",
      source: input.source ?? "web",
      status: "open",
    })
    .returning();

  return row;
}

export async function addTicketMessage(
  input: AddMessageInput,
  db: Db,
): Promise<SupportTicketMessageRow> {
  // Verify ticket belongs to the org
  const [ticket] = await db
    .select({ id: supportTicket.id })
    .from(supportTicket)
    .where(
      and(
        eq(supportTicket.id, input.ticketId),
        eq(supportTicket.organizationId, input.organizationId),
      ),
    )
    .limit(1);

  if (!ticket) {
    throw new Error("NOT_FOUND");
  }

  const [row] = await db
    .insert(supportTicketMessage)
    .values({
      id: crypto.randomUUID(),
      ticketId: input.ticketId,
      organizationId: input.organizationId,
      authorUserId: input.authorUserId,
      channel: input.channel,
      body: input.body,
      isInternal: input.isInternal ?? false,
    })
    .returning();

  return row;
}

export async function getTicket(
  id: string,
  organizationId: string,
  db: Db,
): Promise<SupportTicketRow> {
  const [row] = await db
    .select()
    .from(supportTicket)
    .where(
      and(
        eq(supportTicket.id, id),
        eq(supportTicket.organizationId, organizationId),
      ),
    )
    .limit(1);

  if (!row) {
    throw new Error("NOT_FOUND");
  }

  return row;
}

export async function listOrgTickets(
  organizationId: string,
  filters: ListTicketsFilter,
  db: Db,
): Promise<SupportTicketRow[]> {
  const conditions = [eq(supportTicket.organizationId, organizationId)];
  if (filters.status) conditions.push(eq(supportTicket.status, filters.status));
  if (filters.bookingId) conditions.push(eq(supportTicket.bookingId, filters.bookingId));

  const rows = await db
    .select()
    .from(supportTicket)
    .where(and(...conditions))
    .orderBy(desc(supportTicket.createdAt))
    .limit(filters.limit ?? 50)
    .offset(filters.offset ?? 0);

  return rows;
}
```

```typescript
import z from "zod";

const ticketOutput = z.object({

  customerUserId: z.string().nullable(),
  createdByUserId: z.string().nullable(),
  assignedToUserId: z.string().nullable(),
  source: z.enum(["web", "manual", "api", "email", "telegram"]).nullable(),
  status: z.enum([
    "open",
    "pending_customer",
    "pending_operator",
    "escalated",
    "resolved",
    "closed",
  ]),
  priority: z.enum(["low", "normal", "high", "urgent"]),
  subject: z.string(),
  description: z.string().nullable(),
  createdAt: z.string().datetime(),
  updatedAt: z.string().datetime(),
});

const messageOutput = z.object({
  id: z.string(),
  ticketId: z.string(),
  organizationId: z.string(),
  authorUserId: z.string().nullable(),
  channel: z.enum(["web", "internal", "email"]),
  body: z.string(),
  isInternal: z.boolean(),
  createdAt: z.string().datetime(),
  updatedAt: z.string().datetime(),
});

export const supportContract = {
  createTicket: oc
    .route({ tags: ["Support"], summary: "Create support ticket" })
    .input(
      z.object({
        organizationId: z.string().trim().min(1),
        bookingId: z.string().trim().min(1).optional(),
        subject: z.string().trim().min(1).max(255),
        description: z.string().trim().optional(),
        priority: z.enum(["low", "normal", "high", "urgent"]).optional(),
      }),
    )
    .output(ticketOutput),

  addMessage: oc
    .route({ tags: ["Support"], summary: "Add message to ticket" })
    .input(
      z.object({
        ticketId: z.string().trim().min(1),
        organizationId: z.string().trim().min(1),
        channel: z.enum(["web", "internal", "email"]).default("web"),
        body: z.string().trim().min(1),
        isInternal: z.boolean().optional(),
      }),
    )
    .output(messageOutput),

  getTicket: oc
    .route({ tags: ["Support"], summary: "Get support ticket by ID" })
    .input(z.object({ id: z.string().trim().min(1), organizationId: z.string().trim().min(1) }))
    .output(ticketOutput),

  listOrgTickets: oc
    .route({ tags: ["Support"], summary: "List org support tickets" })

      z.object({
        status: z
          .enum(["open", "pending_customer", "pending_operator", "escalated", "resolved", "closed"])
          .optional(),
        bookingId: z.string().trim().optional(),
        limit: z.number().int().positive().max(100).optional(),
        offset: z.number().int().min(0).optional(),
      }),
    )
    .output(z.array(ticketOutput)),
};
```

```typescript
import { ORPCError } from "@orpc/server";
import { db } from "@my-app/db";
import {
  createSupportTicket,
  addTicketMessage,
  getTicket,
  listOrgTickets,
  type SupportTicketRow,
  type SupportTicketMessageRow,
} from "@my-app/support";

import { organizationPermissionProcedure, protectedProcedure } from "../index";

const formatTicket = (row: SupportTicketRow) => ({
  ...row,
  bookingId: row.bookingId ?? null,
  customerUserId: row.customerUserId ?? null,
  createdByUserId: row.createdByUserId ?? null,
  assignedToUserId: row.assignedToUserId ?? null,
  description: row.description ?? null,
  createdAt: row.createdAt.toISOString(),
  updatedAt: row.updatedAt.toISOString(),
});

const formatMessage = (row: SupportTicketMessageRow) => ({
  ...row,
  authorUserId: row.authorUserId ?? null,
  createdAt: row.createdAt.toISOString(),
  updatedAt: row.updatedAt.toISOString(),
});

export const supportRouter = {
  createTicket: protectedProcedure.support.createTicket.handler(
    async ({ context, input }) => {
      const row = await createSupportTicket(
        {
          organizationId: input.organizationId,
          bookingId: input.bookingId,
          customerUserId: context.session?.user?.id,
          createdByUserId: context.session?.user?.id,
          subject: input.subject,
          description: input.description,
          priority: input.priority,
          source: "web",
        },
        db,
      );
      return formatTicket(row);
    },
  ),

  addMessage: organizationPermissionProcedure({
    support: ["create"],
  }).support.addMessage.handler(async ({ context, input }) => {
    try {
      const row = await addTicketMessage(
        {
          ticketId: input.ticketId,
          organizationId: input.organizationId,
          authorUserId: context.session?.user?.id,
          channel: input.channel,
          body: input.body,
          isInternal: input.isInternal,
        },
        db,
      );
      return formatMessage(row);
    } catch (e) {
      if (e instanceof Error && e.message === "NOT_FOUND") {
        throw new ORPCError("NOT_FOUND");
      }
      throw e;
    }
  }),

  getTicket: organizationPermissionProcedure({
    support: ["read"],
  }).support.getTicket.handler(async ({ input }) => {
    try {
      const row = await getTicket(input.id, input.organizationId, db);
      return formatTicket(row);
    } catch (e) {
      if (e instanceof Error && e.message === "NOT_FOUND") {
        throw new ORPCError("NOT_FOUND");
      }
      throw e;
    }
  }),

  listOrgTickets: organizationPermissionProcedure({
    support: ["read"],
  }).support.listOrgTickets.handler(async ({ context, input }) => {
    const rows = await listOrgTickets(
      context.activeMembership.organizationId,
      {
        status: input.status,
        bookingId: input.bookingId,
        limit: input.limit,
        offset: input.offset,
      },
      db,
    );
    return rows.map(formatTicket);
  }),
};
```

```typescript
// After a successful updateBookingStatus call that returns `row`:
if (
  (input.status === "confirmed" || input.status === "cancelled") &&
  row.customerUserId
) {
  const isConfirmed = input.status === "confirmed";
  const idempotencyKey = `booking.${input.status}:${row.id}`.slice(0, 255);
  await notificationsPusher({
    input: {
      organizationId: row.organizationId,
      actorUserId: context.session?.user?.id,
      eventType: isConfirmed ? "booking.confirmed" : "booking.cancelled",
      sourceType: "booking",
      sourceId: row.id,
      idempotencyKey,
      payload: {
        recipients: [
          {
            userId: row.customerUserId,
            title: isConfirmed ? "Booking confirmed" : "Booking cancelled",
            body: isConfirmed
              ? "Your booking has been confirmed."
              : "Your booking has been cancelled.",
            ctaUrl: `/bookings/${row.id}`,
            channels: ["in_app"],
            severity: isConfirmed ? "success" : "warning",
          },
        ],
      },
    },
    queue: context.notificationQueue,
  });
}
return formatBooking(row);
```

NOTE: The notification call is fire-and-forget from the handler's perspective — the
booking state change has already been committed. If the notification queue is unavailable
(context.notificationQueue is undefined), `notificationsPusher` handles it gracefully.
