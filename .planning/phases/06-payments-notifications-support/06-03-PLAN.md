```typescript
import { eq, and, notInArray, sql, desc } from "drizzle-orm";
import {
  booking,
  bookingCancellationRequest,
  bookingRefund,
  cancellationPolicy,
  organizationSettings,
} from "@my-app/db/schema/marketplace";
import type { Db, CancellationRequestRow, RequestCancellationInput } from "./types";

type CancellationPolicy = {
  freeWindowHours: number;
  penaltyBps: number;
  latePenaltyBps: number;
  latePenaltyWindowHours: number;
};

async function resolveCancellationPolicy(
  organizationId: string,
  listingId: string,
  db: Db,
): Promise<CancellationPolicy> {
  // 1. Try listing-scoped policy first
  const [listingPolicy] = await db
    .select()
    .from(cancellationPolicy)
    .where(
      and(
        eq(cancellationPolicy.listingId, listingId),
        eq(cancellationPolicy.isActive, true),
      ),
    )
    .limit(1);

  if (listingPolicy) {
    return {
      freeWindowHours: listingPolicy.freeWindowHours,
      penaltyBps: listingPolicy.penaltyBps,
      latePenaltyBps: listingPolicy.latePenaltyBps,
      latePenaltyWindowHours: listingPolicy.latePenaltyWindowHours,
    };
  }

  // 2. Try org-scoped policy
  const [orgPolicy] = await db
    .select()
    .from(cancellationPolicy)
    .where(
      and(
        eq(cancellationPolicy.organizationId, organizationId),
        eq(cancellationPolicy.scope, "organization"),
        eq(cancellationPolicy.isActive, true),
      ),
    )
    .limit(1);

  if (orgPolicy) {
    return {
      freeWindowHours: orgPolicy.freeWindowHours,
      penaltyBps: orgPolicy.penaltyBps,
      latePenaltyBps: orgPolicy.latePenaltyBps,
      latePenaltyWindowHours: orgPolicy.latePenaltyWindowHours,
    };
  }

  // 3. Fallback to organizationSettings defaults
  const [settings] = await db
    .select({
      freeWindowHours: organizationSettings.cancellationFreeWindowHours,
      penaltyBps: organizationSettings.cancellationPenaltyBps,
    })
    .from(organizationSettings)
    .where(eq(organizationSettings.organizationId, organizationId))
    .limit(1);

  return {
    freeWindowHours: settings?.freeWindowHours ?? 24,
    penaltyBps: settings?.penaltyBps ?? 0,
    latePenaltyBps: 10_000,
    latePenaltyWindowHours: 2,
  };
}

function computePenalty(
  totalPriceCents: number,
  startsAt: Date,
  policy: CancellationPolicy,
): { penaltyAmountCents: number; refundAmountCents: number } {
  const now = Date.now();
  const hoursUntilStart = (startsAt.getTime() - now) / (1000 * 60 * 60);

  let penaltyAmountCents: number;
  if (hoursUntilStart >= policy.freeWindowHours) {
    penaltyAmountCents = 0;
  } else if (hoursUntilStart < policy.latePenaltyWindowHours) {
    penaltyAmountCents = Math.round(
      (totalPriceCents * policy.latePenaltyBps) / 10_000,
    );
  } else {
    penaltyAmountCents = Math.round(
      (totalPriceCents * policy.penaltyBps) / 10_000,
    );
  }

  return {
    penaltyAmountCents,
    refundAmountCents: totalPriceCents - penaltyAmountCents,
  };
}

export async function requestCancellation(
  input: RequestCancellationInput,
  db: Db,
): Promise<CancellationRequestRow> {
  // 1. Load booking
  const [row] = await db
    .select()
    .from(booking)
    .where(
      and(
        eq(booking.id, input.bookingId),
        eq(booking.organizationId, input.organizationId),
      ),
    )
    .limit(1);

  if (!row) throw new Error("NOT_FOUND");

  // 2. Guard invalid booking states
  const cancellableStatuses = ["pending", "awaiting_payment", "confirmed"] as const;
  if (!cancellableStatuses.includes(row.status as typeof cancellableStatuses[number])) {
    throw new Error("INVALID_STATE");
  }

  // 3. Guard duplicate active request
  const terminalStatuses = ["rejected", "cancelled"] as const;
  const existing = await db
    .select({ id: bookingCancellationRequest.id })
    .from(bookingCancellationRequest)
    .where(
      and(
        eq(bookingCancellationRequest.bookingId, input.bookingId),
        notInArray(bookingCancellationRequest.status, [...terminalStatuses]),
      ),
    )
    .limit(1);

  if (existing.length > 0) throw new Error("DUPLICATE_REQUEST");

  // 4. Resolve policy and compute financials
  const policy = await resolveCancellationPolicy(
    input.organizationId,
    row.listingId,
    db,
  );
  const { penaltyAmountCents, refundAmountCents } = computePenalty(
    row.totalPriceCents,
    row.startsAt,
    policy,
  );

  // 5. Insert cancellation request
  const [request] = await db
    .insert(bookingCancellationRequest)
    .values({
      id: crypto.randomUUID(),
      bookingId: input.bookingId,
      organizationId: input.organizationId,
      requestedByUserId: input.requestedByUserId,
      initiatedByRole: input.initiatedByRole,
      status: "requested",
      reason: input.reason,
      bookingTotalPriceCents: row.totalPriceCents,
      penaltyAmountCents,
      refundAmountCents,
      currency: row.currency,
    })
    .returning();

  return request;
}

export async function applyCancellation(
  requestId: string,
  organizationId: string,
  approvedByUserId: string,
  db: Db,
): Promise<{ request: CancellationRequestRow; refundId: string | null }> {
  // 1. Load and validate the request
  const [request] = await db
    .select()
    .from(bookingCancellationRequest)
    .where(
      and(
        eq(bookingCancellationRequest.id, requestId),
        eq(bookingCancellationRequest.organizationId, organizationId),
      ),
    )
    .limit(1);

  if (!request) throw new Error("NOT_FOUND");
  if (request.status !== "requested") throw new Error("INVALID_STATE");

  const now = new Date();
  let refundId: string | null = null;

  // 2. Apply atomically
  const result = await db.transaction(async (tx) => {
    // Update cancellation request
    const [updatedRequest] = await tx
      .update(bookingCancellationRequest)
      .set({
        status: "applied",
        appliedByUserId: approvedByUserId,
        appliedAt: now,
        updatedAt: sql`now()`,
      })
      .where(eq(bookingCancellationRequest.id, requestId))
      .returning();

    // Cancel the booking
    await tx
      .update(booking)
      .set({
        status: "cancelled",
        cancelledAt: now,
        cancelledByUserId: approvedByUserId,
        cancellationReason: request.reason,
        refundAmountCents: request.refundAmountCents,
        paymentStatus: request.refundAmountCents > 0 ? "refunded" : undefined,
        updatedAt: sql`now()`,
      })
      .where(eq(booking.id, request.bookingId));

    // Insert refund row if there is a non-zero refund
    let insertedRefundId: string | null = null;
    if (request.refundAmountCents > 0) {
      insertedRefundId = crypto.randomUUID();
      await tx.insert(bookingRefund).values({
        id: insertedRefundId,
        bookingId: request.bookingId,
        organizationId,
        requestedByUserId: approvedByUserId,
        approvedByUserId,
        status: "requested",
        amountCents: request.refundAmountCents,
        currency: request.currency,
      });
    }

    return { updatedRequest, refundId: insertedRefundId };
  });

  refundId = result.refundId;

  return { request: result.updatedRequest, refundId };
}

export async function getActiveCancellationRequest(
  bookingId: string,
  organizationId: string,
  db: Db,
): Promise<CancellationRequestRow | null> {
  const terminalStatuses = ["rejected", "cancelled"] as const;
  const [row] = await db
    .select()
    .from(bookingCancellationRequest)
    .where(
      and(
        eq(bookingCancellationRequest.bookingId, bookingId),
        eq(bookingCancellationRequest.organizationId, organizationId),
        notInArray(bookingCancellationRequest.status, [...terminalStatuses]),
      ),
    )
    .limit(1);

  return row ?? null;
}

export async function listOrgCancellationRequests(
  organizationId: string,
  db: Db,
): Promise<CancellationRequestRow[]> {
  return db
    .select()
    .from(bookingCancellationRequest)
    .where(eq(bookingCancellationRequest.organizationId, organizationId))
    .orderBy(desc(bookingCancellationRequest.requestedAt));
}
```
